sudo nano /etc/systemd/system/gunicorn.service

[Unit]
Description=znasapi gunicorn daemon
Requires=gunicorn.socket
After=network.target

[Service]
User=znas
Group=www-data
WorkingDirectory=/home/znas/znasapi
ExecStart=/home/znas/znasapi/zenv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/run/gunicorn.sock \
          project.wsgi:application

[Install]
WantedBy=multi-user.target


==========================================================================================

upstream znasapi_server {
    server unix:/run/gunicorn.sock fail_timeout=0;
}

server { 
    server_name api.zenv.in;
    client_max_body_size 100M;
    charset utf-8;
    underscores_in_headers on;
    keepalive_timeout 5;
    proxy_ignore_client_abort on;

    error_log /home/znas/znasapi/logs/nginx-error.log;

    location = /favicon.png { access_log off; log_not_found off; }
    location /static/ {
        alias /home/znas/znasapi/static/staticfiles/;
    }
    location /media/ {
        root /home/znas/znasapi;
        client_max_body_size 100M;
    }
    location / {
        include proxy_params;    
        add_header 'Access-Control-Allow-Origin' '*';
        add_header 'Access-Control-Allow-Headers' 'authorization, content-type';
        add_header 'X-Forwarded-Host' '*';    
        proxy_pass http://znasapi_server;
    }
}
